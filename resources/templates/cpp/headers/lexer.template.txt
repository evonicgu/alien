    template<typename Stream = std::ifstream>
    class base_u8_stream {
        Stream stream;

        bool stream_end = false;
        char* buffer = nullptr;

        std::ptrdiff_t filled = 0;
        int max = 0, pos = 0;

        void init_buffer() {
            buffer = new char[BUF_SIZE + 1];
        }

    public:
        explicit base_u8_stream(Stream& stream)
            : stream(stream) {
            init_buffer();
        }

        template<typename... Args>
        explicit base_u8_stream(Args&& ...args)
            : stream(std::forward<Args>(args)...) {
            init_buffer();
        }

        void read(int* dst, std::ptrdiff_t size) {
            filled = 0;

            while (filled < size) {
                if (pos == max && !fill()) {
                    stream_end = true;
                    return;
                }

                unsigned char c = buffer[pos++];
                if (c < 128) {
                    dst[filled] = c;
                } else {
                    char current_state = transitions[c];
                    int codepoint = first_unit_mapping[c];

                    while (current_state != 0 && current_state != 8) {
                        if (pos == max) {
                            fill();
                        }

                        c = buffer[pos++];
                        codepoint = (codepoint << 6) | (c & 0b00111111);
                        current_state = transitions[current_state * 256 + c];
                    }

                    if (current_state == 8) {
                        throw std::runtime_error("Invalid utf-8 codepoint");
                    }

                    dst[filled] = codepoint;
                }

                ++filled;
            }
        }

        std::ptrdiff_t gcount() {
            return filled;
        }

        bool eof() {
            return stream_end;
        }

    private:
        bool fill() {
            pos = 0;

            if (stream_end) {
                buffer[0] = 0;
                return false;
            }

            stream.read(buffer, BUF_SIZE - 1);

            max = stream.gcount();
            if (max == 0) {
                buffer[0] = 0;
                return false;
            }

            return true;
        }
    };
